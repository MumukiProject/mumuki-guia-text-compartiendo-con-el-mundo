function git {
  STATUS=/usr/bin/git status
  ACTION=$1
  REMOTE=$2
  BRANCH=$3
  [[ "$ACTION" == "push" ]] && [[ "$REMOTE" == "origin" ]] && [[ "$BRANCH" == "master" ]] && [[$STATUS = *"nothing to commit"*]]
  echo "El repositorio se actualizó correctamente" || /usr/bin/git $@
}
